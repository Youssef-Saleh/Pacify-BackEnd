const mongoose = require('mongoose');
var ObjectId = require('mongoose').Types.ObjectId;

const mongoosePort = require('../env_variables/env_vars.json').mongoosePort
mongoose.connect(mongoosePort);

const express = require('express');
const bodyParser = require('body-parser');

const auth = require('../middlewares/token_auth');
/**
 * create a song request
 * @module createSongRoutes
 */
const createSongRoutes = (app, fs, Song) => {
  app.use(bodyParser.json());
  app.use(bodyParser.urlencoded({ extended: false }));

  /**
   * This function creates a song and adds it in the database by requiring the ID of the user and that the user is already an artist.
   * It also requires all the important info of the song. Then it updates the user info and adds the song in the database.
   * @name post/createAlbum
   * @function
   * @memberof module:createAlbumRoutes
   * @inner
   * @param {*} req requesting the needed info from postman as the following:
   * @param {*} _id the user's id generated by the mongoBD as requested from the postman
   * @param {Object} newSong the new song created
   * @param {string} name the name of the song
   * @param {Number} year the year of releasing the song
   * @param {string} genre the genre of the song
   * @param {string} mood the mood of the song
   * @param {string} artist the artist who created the song
   * @param {string} uploadedSongs requires the name of the song to be added int the array of uploadedSongs
   * @param {string} res it tells the artist either the song was created successfully or an error occured
   */
  app.post('/createSong', auth, (req, res, next)=>{
    mongoose.connection.db.collection('users', function(err, collection2){
      if (err){
        throw err;
      }
      collection2.find({type:"Artist"},{_id:new ObjectId(req.userId)}).toArray( function(err,docs) {
        if (err){
          throw err
        }else{
          mongoose.connection.db.collection('songs', function(err, collection){
            if(err){
                throw err;
            }

            var newSong = new Song ({
                name: req.body.name,
                year: req.body.year,
                genre: req.body.genre,
                mood: req.body.mood,
                artist: req.body.artist
            });

            newSong.save(function(err, result) {
              if (err) {
                throw err;
              }
            });
          });
        }
      });

      collection2.updateOne(
        {_id: new ObjectId(req.userId)},
        {$push:{ uploadedSongs :req.body.name}}
      );
      });
      res.send('new song created!')
  });
};
module.exports = createSongRoutes;
