const mongoose = require('mongoose');
var ObjectId = require('mongoose').Types.ObjectId;

const mongoosePort = require('../env_variables/env_vars.json').mongoosePort
mongoose.connect(mongoosePort);

const express = require('express');
const bodyParser = require('body-parser');
const auth = require('../middlewares/token_auth');

var Album = require('../Database Seeds/models/album');

 /**
 * create an album request
 * @module createAlbumRoutes
 */
const createAlbumRoutes = (app, fs) => {
    app.use(bodyParser.json());
    app.use(bodyParser.urlencoded({ extended: false }));
    /**
     * This function creates an album and adds it in the database by requiring the ID of the user and that the user is already an artist.
     * It also requires all the important info of the album. Then it updates the user info and adds the album in the database.
     * @name post/createAlbum
     * @function
     * @memberof module:createAlbumRoutes
     * @inner
     * @param {*} req requesting the needed info from postman as the following:
     * @param {*} _id the user's id generated by the mongoBD as requested from the postman
     * @param {Object} nAlbum the new album created.
     * @param {string} imagePath the link of the image of the album
     * @param {string} name the name of the album
     * @param {Number} year the year of releasing the album
     * @param {string} uploadedAlbums requires the name of the album to be added int the array of uploadedAlbums
     * @param {string} res it tells the artist either the album was created successfully or an error occured
     */

    app.post('/createAlbum', auth, (req, res, next)=>{

        mongoose.connection.db.collection('users', function(err, collection2){
            if (err){ 
                throw err;
            }
            collection2.find({_id:new ObjectId(req.userId)}, {type:"Artist"}).toArray( function(err,docs) {

                if (err){
                    throw err
                }else{
                    mongoose.connection.db.collection('albums', function(err, collection){
                        if(err){
                            throw err;
                        }
                        
                        var nAlbum = new Album ({
                            imagePath: req.body.imagePath,
                            name: req.body.name,
                            year: req.body.year,
                        });

                        nAlbum.save(function(err, result) {
                            if (err) {
                              throw err;
                            }
                        });
                    });
                }
                collection2.updateOne(
                    {_id: new ObjectId(req.userId)},
                    {$push:{ uploadedAlbums :req.body.name}}
                );
            });
        });
        res.send('new album created!')
    });
};
module.exports = createAlbumRoutes;
